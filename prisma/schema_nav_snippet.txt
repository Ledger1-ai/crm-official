
model NavigationConfig {
  id        String   @id @default(auto()) @map("_id") @db.ObjectId
  team_id   String   @db.ObjectId
  user_id   String?  @db.ObjectId // Optional: If set, it's a specific user preference override
  
  // The structure of the sidebar
  // Format: Array of { id, type: 'item'|'group', label, icon, href, children: [], permissions: [], hidden: boolean }
  structure Json     

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  assigned_team Team   @relation(fields: [team_id], references: [id])
  assigned_user Users? @relation(fields: [user_id], references: [id])

  @@index([team_id, user_id])
}
